#version 330
uniform vec3 vertex1;
uniform vec3 vertex2;
uniform vec3 vertex3;
void main(){ if (gl_VertexID == 0) {
        gl_Position = vec4(vertex1, 1.0);
    } else if (gl_VertexID == 1) {
        gl_Position = vec4(vertex2, 1.0);
    } else if (gl_VertexID == 2) {
        gl_Position = vec4(vertex3, 1.0);
    }
};
